cmake_minimum_required(VERSION 3.16)
project(golanglint VERSION 1.0 LANGUAGES C CXX)

set(CMAKE_INCLUDE_CURRENT_DIR ON)

# Set up AUTOMOC and some sensible defaults for runtime execution
# When using Qt 6.3, you can replace the code block below with
# qt_standard_project_setup()
set(CMAKE_AUTOMOC ON)
include(GNUInstallDirs)
set(CMAKE_AUTOUIC ON)

find_package(QT NAMES Qt5 Qt6 REQUIRED COMPONENTS Core)
find_package(Qt${QT_VERSION_MAJOR} REQUIRED COMPONENTS Gui)
find_package(Qt${QT_VERSION_MAJOR} OPTIONAL_COMPONENTS PrintSupport Widgets)

qt_add_plugin(golanglint)
target_sources(golanglint PRIVATE
    golanglint.cpp golanglint.h
    golanglint_global.h
    golanglintoption.cpp golanglintoption.h golanglintoption.ui
    golanglintoptionfactory.cpp golanglintoptionfactory.h
    golanglintplugin.cpp golanglintplugin.h
)
